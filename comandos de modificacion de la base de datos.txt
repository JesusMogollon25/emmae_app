-- Obtener nombre del responsable (estudiante o docente) de un instrumento
SELECT i.id_instrumento,
CASE
WHEN i.id_estudiante IS NOT NULL THEN CONCAT(e.nombre,' ',e.apellido)
WHEN i.id_docente IS NOT NULL THEN CONCAT(d.nombre,' ',d.apellido)
ELSE 'Sin responsable'
END AS responsable
FROM instrumentos i
LEFT JOIN estudiantes e ON i.id_estudiante = e.id_estudiante
LEFT JOIN docente d ON i.id_docente = d.id_docente;
